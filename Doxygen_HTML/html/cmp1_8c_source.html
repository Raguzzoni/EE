60void CMP1_Initialize(void)
61{
62 /* Disable CMP1 interrupt before configuring the CMP1
63 control register. Although a comparator is disabled, an interrupt can be
64 generated by changing the output polarity with the CxPOL bit of the
65 CMxCON0 register,or by switching the comparator on or off with the CxON
66 bit of the CMxCON0 register.*/
67 PIE2bits.C1IE = 0;
68
69 // C1HYS disabled; C1SP hi_speed; C1ON enabled; C1POL not inverted; C1OE COUT_internal; C1SYNC asynchronous;
70 CM1CON0 = 0x84;
71
72 // C1INTN no_intFlag; C1INTP intFlag_pos; C1PCH FVR; C1NCH CIN0-;
73 CM1CON1 = 0xA0;
74
75 // Clearing IF flag before enabling the interrupt.
76 PIR2bits.C1IF = 0;
77
78 // Enabling CMP1 interrupt.
79 PIE2bits.C1IE = 1;
80}
