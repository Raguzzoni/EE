64void TMR1_Initialize(void)
65{
66 //Set the Timer to the options selected in the GUI
67
68 //T1GSS T1G_pin; TMR1GE disabled; T1GTM disabled; T1GPOL low; T1GGO done; T1GSPM disabled;
69 T1GCON = 0x00;
70
71 //TMR1H 21;
72 TMR1H = 0x15;
73
74 //TMR1L 160;
75 TMR1L = 0xA0;
76
77 // Clearing IF flag before enabling the interrupt.
78 PIR1bits.TMR1IF = 0;
79
80 // Load the TMR value to reload variable
81 timer1ReloadVal=(uint16_t)((TMR1H << 8) | TMR1L);
82
83 // Enabling TMR1 interrupt.
84 PIE1bits.TMR1IE = 1;
85
86 // Set Default Interrupt Handler
87 TMR1_SetInterruptHandler(TMR1_DefaultInterruptHandler);
88
89 // T1CKPS 1:1; T1OSCEN disabled; nT1SYNC synchronize; TMR1CS FOSC/4; TMR1ON enabled;
90 T1CON = 0x01;
91}

120void TMR1_WriteTimer(uint16_t timerVal)
121{
122 if (T1CONbits.nT1SYNC == 1)
123 {
124 // Stop the Timer by writing to TMRxON bit
125 T1CONbits.TMR1ON = 0;
126
127 // Write to the Timer1 register
128 TMR1H = (uint8_t)(timerVal >> 8);
129 TMR1L = (uint8_t)timerVal;
130
131 // Start the Timer after writing to the register
132 T1CONbits.TMR1ON =1;
133 }
134 else
135 {
136 // Write to the Timer1 register
137 TMR1H = (uint8_t)(timerVal >> 8);

157void TMR1_ISR(void)
158{
159 static volatile unsigned int CountCallBack = 0;
160
161 // Clear the TMR1 interrupt flag
162 PIR1bits.TMR1IF = 0;
163 TMR1_WriteTimer(timer1ReloadVal);
164
165 // callback function - called every 10th pass
166 if (++CountCallBack >= TMR1_INTERRUPT_TICKER_FACTOR)
167 {
168 // ticker function call
169 TMR1_CallBack();
170
171 // reset ticker counter
172 CountCallBack = 0;
173 }
