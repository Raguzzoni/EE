95void Update_tx_buffer(){
96 var.A = current_floor;
97 var.D = next_floor;
98 if(MOTOR_ON && MOTOR_DIR==0) var.M = 2;
99 else if(MOTOR_ON && MOTOR_DIR==1) var.M=3;
100 else var.M=0;
101 if(TMR0_ReadTimer() < 3) TMR0_Reload();
102 position0 = var.HHH;
103 if(MOTOR_DIR != 0) var.HHH += (TMR0_ReadTimer()*1655)/1000;
104 if(MOTOR_DIR == 0) var.HHH -= (TMR0_ReadTimer()*1655)/1000;
105 TMR0_Reload();
106 if(var.HHH >= position0) var.VVV = ((var.HHH - position0)*100/3)&0x01FF;
107 else var.VVV = ((position0 - var.HHH)*100/3)&0x01FF;
108
109
110 ADCON0bits.GO_nDONE = 1; // inicia conversão
111 while (ADCON0bits.GO_nDONE); // espera terminar
112 var.TTT = ((ADRESH << 8) + ADRESL);
113
114 uart_write_buffer[1] = var.A + 1 + '0';
115 uart_write_buffer[3] = var.D + 1 + '0';
116 uart_write_buffer[5] = var.M + '0';
117 uart_write_buffer[7] = (var.HHH / 100) % 10 + '0'; // centena
118 uart_write_buffer[8]= (var.HHH/ 10) % 10 + '0'; // dezena
119 uart_write_buffer[9]=var.HHH % 10 + '0';
120 uart_write_buffer[11]= (var.VVV/100)%10 + '0'; // dezena
121 uart_write_buffer[12]= ((var.VVV/10)%10) + '0';
122 uart_write_buffer[14]= (var.VVV)%10 + '0';
123 uart_write_buffer[16]= (var.TTT/100)%10 + '0';
124 uart_write_buffer[17]= (var.TTT/10)%10 + '0';
125 uart_write_buffer[19]= (var.TTT%10) + '0';
