82void PWM3_LoadDutyValue(uint16_t dutyValue)
83{
84 // Writing to 8 MSBs of pwm duty cycle in CCPRL register
85 CCPR3L = ((dutyValue & 0x03FC)>>2);
86
87 // Writing to 2 LSBs of pwm duty cycle in CCPCON register
88 CCP3CON = ((uint8_t)(CCP3CON & 0xCF) | ((dutyValue & 0x0003)<<4));
89}
