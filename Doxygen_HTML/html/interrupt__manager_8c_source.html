52void __interrupt() INTERRUPT_InterruptManager (void)
53{
54 // interrupt handler
55 if(INTCONbits.IOCIE == 1 && INTCONbits.IOCIF == 1)
56 {
57 PIN_MANAGER_IOC();
58 }
59 if(INTCONbits.PEIE == 1)
60 {
61 if(PIE1bits.ADIE == 1 && PIR1bits.ADIF == 1)
62 {
63 ADC_ISR();
64 }
65 if(PIE1bits.RCIE == 1 && PIR1bits.RCIF == 1)
66 {
67 EUSART_RxDefaultInterruptHandler();
68 }
69 if(PIE2bits.C1IE == 1 && PIR2bits.C1IF == 1)
70 {
71 CMP1_ISR();
72 }
73 if(PIE2bits.C2IE == 1 && PIR2bits.C2IF == 1)
74 {
75 CMP2_ISR();
76 }
77 if(PIE1bits.TMR1IE == 1 && PIR1bits.TMR1IF == 1)
78 {
79 TMR1_ISR();
80 }
81 }
82}
