83void EUSART_Initialize(void)
84{
85 // disable interrupts before changing states
86 PIE1bits.RCIE = 0;
87 EUSART_SetRxInterruptHandler(EUSART_Receive_ISR);
88 // Set the EUSART module to the options selected in the user interface.
89
90 // ABDOVF no_overflow; SCKP Non-Inverted; BRG16 16bit_generator; WUE disabled; ABDEN disabled;
91 BAUDCON = 0x08;
92
93 // SPEN enabled; RX9 8-bit; CREN enabled; ADDEN disabled; SREN disabled;
94 RCSTA = 0x90;
95
96 // TX9 8-bit; TX9D 0; SENDB sync_break_complete; TXEN enabled; SYNC asynchronous; BRGH hi_speed; CSRC master;
97 TXSTA = 0xA4;
98
99 // SPBRGL 103;
100 SPBRGL = 0x67;
101
102 // SPBRGH 0;
103 SPBRGH = 0x00;

107 EUSART_SetOverrunErrorHandler(EUSART_DefaultOverrunErrorHandler);

173void EUSART_Receive_ISR(void)
174{
175
176 eusartRxStatusBuffer[eusartRxHead].status = 0;
177
178 if(RCSTAbits.FERR){
179 eusartRxStatusBuffer[eusartRxHead].ferr = 1;
180 EUSART_FramingErrorHandler();
181 }
182
183 if(RCSTAbits.OERR){
184 eusartRxStatusBuffer[eusartRxHead].oerr = 1;
185 EUSART_OverrunErrorHandler();
186 }
187
188 if(eusartRxStatusBuffer[eusartRxHead].status){
189 EUSART_ErrorHandler();
190 } else {
191 EUSART_RxDataHandler();
192 }
193
194 // or set custom function using EUSART_SetRxInterruptHandler()
195}
